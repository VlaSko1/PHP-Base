<?php
    $a = 5;
    $b = '05';
    var_dump($a == $b);         // Почему true? В данном случае происходит неявное приведение типов, строка преобзаруется в число и получается 5, т.е. равенство.
    var_dump((int)'012345');     // Почему 12345? Здесь происходит явное приведение типа к числу, получаем число 12345
    var_dump((float)123.0 === (int)123.0); // Почему false? Оператор "===" возвращает false в том числе, если типы значений разные. В данном случае имеем два разных типа.
    var_dump((int)0 === (int)'hello, world'); // Почему true? Происходит явное приведение типов к целому числу, строка в PHP при приведении к числу равна 0 (если не начинается с цифры), поэтому 0 === 0 - true.
?>
